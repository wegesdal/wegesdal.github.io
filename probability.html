rolls = []
count_evens = 0
count_over_eighteen = 0

while len(rolls) < 101:
    rolls.append(random.randint(1, 6)*random.randint(1, 6))
for item in rolls:
    if item % 2 == 0:
        count_evens += 1
    if item < 18:
        count_over_eighteen += 1
print(count_evens)
print(count_over_eighteen)

app = App(width=320, height=320)

while len(rolls) < 100:
    rolls.append(random.randint(1, 6) * random.randint(1, 6))
    
@app.event
def on_draw():
    count_a = 0
    count_b = 0
    x = 16
    y = 16
    app.window.clear()
    for item in rolls:
        if item % 2 == 0:
            count_a += 1
            wink.x = x
            wink.y = y
            wink.draw()
        else:
            frown.x = x
            frown.y = y
            frown.draw()
        x += 32
        if x > 320:
            x = 16
            y += 32
        if item < 18:
            count_b += 1

app.run()
